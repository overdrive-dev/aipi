+/**
+ * @file admin-gestores-nav.ts
+ * @ticket NORA-220 (run 20260622T221639Z-c5eca4 review · Finding 2) — gestor↔
+ *         coordenador 1st-click desync, non-sidebar callers.
+ *
+ * The `/gestores` list page is URL-authoritative for `?tipo` (the admin-nav
+ * store is only a fallback for when the query is dropped — see
+ * `app/(app)/(admin)/gestores/index.tsx` and `utils/web-query-param.ts`).
+ *
+ * `AdminSidebar.handleNavigation` already seeds the store AND materialises the
+ * full `?tipo` URL on the SAME TICK. But every OTHER caller that navigates to a
+ * gestor/coordenador route (invite success/cancel, breadcrumb back-links,
+ * user-detail back/delete) used a plain string `router.push`/`router.replace`,
+ * which on expo-router web can DROP the `?tipo` query. When the query is lost,
+ * `/gestores` falls back to a possibly-STALE persisted store and renders the
+ * wrong bucket — e.g. cancelling a coordinator invite landing on "Gestores
+ * internos" (the exact regression this fix closes).
+ *
+ * These pure helpers centralise the same-tick contract used by the sidebar so
+ * EVERY gestor/coordenador navigation can:
+ *   1. seed the store synchronously to the destination bucket (so even if the
+ *      query is dropped, the fallback is now CORRECT instead of stale), and
+ *   2. materialise the full destination URL (path + `?tipo`) on web on the SAME
+ *      TICK via the History API — no timer, no race, no hard reload.
+ *
+ * Kept free of any React-Native / store import so the resolver + URL builder
+ * stay unit-testable in a plain Node (ts-jest) environment. The store write
+ * lives in the `useGestoresNav` hook that composes these.
+ */
+import type { GestoresTipo } from '@/stores/admin-nav-store';
+
+export type { GestoresTipo };
+
+/**
+ * Resolve which `/gestores` bucket a route targets.
+ *
+ * - explicit `?tipo=coordenadores` / `?tipo=internos` → that bucket
+ * - the canonical gestores list/invite base with no `?tipo` → `internos`
+ *   (the gestores surface defaults to internos)
+ * - anything else (e.g. `/profissionais`, `/visualizadores`) → `undefined`
+ *   so callers leave the store untouched for non-gestores destinations.
+ */
+export function resolveGestoresTipo(route: string): GestoresTipo | undefined {
+  if (/[?&]tipo=coordenadores\b/.test(route)) return 'coordenadores';
+  if (/[?&]tipo=internos\b/.test(route)) return 'internos';
+  // The canonical gestores list/invite base (no ?tipo) means "internos".
+  if (/\/gestores(\/convidar)?(\?|$)/.test(route)) return 'internos';
+  return undefined;
+}
+
+/**
+ * Build the web address-bar URL for a route by stripping Expo Router group
+ * segments (`(app)`, `(admin)`), which never appear in the browser URL.
+ * The path + any query (`?tipo=...`) are preserved.
+ */
+export function gestoresWebUrl(route: string): string {
+  const stripped = route.replace(/\/\([^)]*\)/g, '');
+  return stripped.startsWith('/') ? stripped : `/${stripped}`;
+}
+
+/**
+ * Materialise the full gestor/coordenador URL (path + `?tipo`) into the address
+ * bar on web, on the SAME TICK, with no reload. No-op on native, when the
+ * History API is unavailable, when the route is not a gestores route, or when
+ * the URL is already correct. Mirrors the sidebar's `replaceState` contract.
+ */
+export function materializeGestoresUrlOnWeb(route: string): void {
+  // Only act on gestores routes — leave /profissionais, /visualizadores, etc.
+  if (resolveGestoresTipo(route) === undefined) return;
+  if (
+    typeof window === 'undefined' ||
+    typeof window.history?.replaceState !== 'function'
+  ) {
+    return;
+  }
+  const webUrl = gestoresWebUrl(route);
+  const currentUrl = `${window.location.pathname}${window.location.search}`;
+  if (currentUrl !== webUrl) {
+    window.history.replaceState(window.history.state, '', webUrl);
+  }
+}
